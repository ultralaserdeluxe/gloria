\section{Sensormodul}
\subsection{Inledande beskrivning av sensormodul}
Sensormodulen sköter sensordata för linjeföljare, avståndsmätare för detektering av objekt samt eventuella andra sensorer som kan behövas. 

\subsection{Gränssnitt}
\begin{LIPSkravlista}
\LIPSkrav{Orginal}{Sensormodulen skall skicka och ta emot nödvändiga data från huvudmodulen}{1}
\end{LIPSkravlista}

\subsection{Funktionella krav för sensormodul}
\begin{LIPSkravlista}
\LIPSkrav{Orginal}{Sensorerna skall kunna kalibreras}{1}
\LIPSkrav{Orginal}{Sensorer skall kunna detektera banan}{1}
\LIPSkrav{Orginal}{Sensorer skall kunna detektera ett paket}{1}
\LIPSkrav{Orginal}{Sensorer skall kunna identifiera RFID}{1}
\end{LIPSkravlista}
